frequency array with shifting 


#include<bits/stdc++.h>
#include <iostream>

using namespace std;

 int main() {
int n ;
cin >> n;
int a[n] ;
 for(int i = 0; i < n ; i++){
    cin >> a[i] ;
    }
    int mx=*max_element(a,a+n);
    int shift=abs(*min_element(a,a+n));
    int freq[mx+shift+1]={0};
    for (int i=0;i<n;i++){
        freq[a[i]+shift]++;
    }
